def all(self, cls=None):
	classes = [User, State, City, Amenity, Place, Review]
	dictionary = {}

	if cls is None:
		for item in classes:
		query = self.__session.query(item).all()
		for obj in query:
			objClass= obj.__class__.__name__
			objId = obj.id
			objKey = str(objClass)+'.'+str(objId)

			dictionary[objKey] = obj

		return dictionary
	else:
		our_query = self.__session.query(cls).all()

		for obj in our_query:
		objClass= obj.__class__.__name__
		objId = obj.id
		objKey = str(objClass)+'.'+str(objId)

		dictionary[objKey] = obj
		return dictionary
